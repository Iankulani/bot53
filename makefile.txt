# BOT53 Makefile

.PHONY: help install run docker-build docker-run clean test lint

# Variables
VENV = bot53_env
PYTHON = $(VENV)/bin/python3
PIP = $(VENV)/bin/pip

help:
	@echo "Available commands:"
	@echo "  make install     - Install dependencies"
	@echo "  make run         - Run BOT53"
	@echo "  make docker-build - Build Docker image"
	@echo "  make docker-run  - Run with Docker"
	@echo "  make clean       - Clean up"
	@echo "  make test        - Run tests"
	@echo "  make lint        - Lint code"

install:
	@echo "Creating virtual environment..."
	python3 -m venv $(VENV)
	@echo "Installing dependencies..."
	$(PIP) install --upgrade pip
	$(PIP) install -r requirements.txt
	@echo "Installing additional tools..."
	chmod +x install.sh
	./install.sh

run:
	@echo "Starting BOT53..."
	@if [ ! -d "$(VENV)" ]; then \
		echo "Virtual environment not found. Running install..."; \
		make install; \
	fi
	$(PYTHON) bot53.py

docker-build:
	@echo "Building Docker image..."
	docker build -t bot53:latest .
	@echo "Image built: bot53:latest"

docker-run:
	@echo "Running with Docker..."
	docker run -it --rm --privileged --network host \
		-v bot53_data:/home/bot53/.bot53 \
		-v bot53_reports:/home/bot53/bot53_reports \
		bot53:latest

docker-compose:
	@echo "Running with Docker Compose..."
	docker-compose up --build

clean:
	@echo "Cleaning up..."
	rm -rf $(VENV)
	rm -rf __pycache__
	rm -rf *.pyc
	rm -rf .pytest_cache
	rm -rf .coverage
	rm -rf coverage.xml
	rm -rf .tox
	@echo "Clean complete!"

test:
	@echo "Running tests..."
	$(PYTHON) -m pytest tests/ -v --cov=bot53

lint:
	@echo "Linting code..."
	$(VENV)/bin/flake8 bot53.py --max-line-length=120
	$(VENV)/bin/pylint bot53.py --exit-zero

security:
	@echo "Security scanning..."
	$(VENV)/bin/safety check -r requirements.txt
	$(VENV)/bin/bandit -r . -f json -o bandit-report.json

all: install test lint run
